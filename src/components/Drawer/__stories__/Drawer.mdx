## Usage

Below is an example of how to use the Drawer component:

```jsx
  import React, { useRef } from 'react';
  import { Drawer } from 'drawer-component-library';

  const MyComponent = () => {
    const drawerRef = useRef();

    const handleOpen = () => {
      drawerRef.current?.open();
    };

    const handleClose = () => {
      drawerRef.current?.close();
    };

    return (
     <div>
        <div onClick={handleOpen}>
          Open Drawer
       </div>
       <Drawer ref={drawerRef} openHeight={50}>
          <h4>Drawer contents...</h4>
      </div>
    );
  };
```
